```python
from math import sqrt
a, b, c = map(int, input().split())
if b*b - 4*a*c > 0:
    temp = sqrt(b*b-4*a*c)
    x1 = (-b+temp) / (2*a)
    x2 = (-b-temp) / (2*a)
    print("Two different roots x1=", int(x1), " , x2=", int(x2), sep='')
elif b*b-4*a*c == 0:
    temp = sqrt(b*b - 4*a*c)
    x1 = (-b) / (2*a);
    print("Two same roots x=", int(x1), sep='')
else:
    print("No real root")
```
